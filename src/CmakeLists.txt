cmake_minimum_required(VERSION 3.10)

# set the project name
project(ActiveFanControl VERSION 0.1)

# specify the C++ standard
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

option(USE_CONFIG "Use CONFIG" ON)
option(USE_THERMAL "Use THERMAL" ON)
option(USE_PWM "Use PWM" ON)
option(USE_FAN "Use FAN" ON)

if(USE_CONFIG)
  add_subdirectory(config)
  list(APPEND EXTRA_LIBS config)
  list(APPEND EXTRA_INCLUDES "${PROJECT_SOURCE_DIR}/config")
endif()

if(USE_THERMAL)
  add_subdirectory(thermal)
  list(APPEND EXTRA_LIBS thermal)
  list(APPEND EXTRA_INCLUDES "${PROJECT_SOURCE_DIR}/thermal")
endif()

if(USE_PWM)
  add_subdirectory(pwm)
  list(APPEND EXTRA_LIBS pwm)
  list(APPEND EXTRA_INCLUDES "${PROJECT_SOURCE_DIR}/pwm")
endif()

if(USE_FAN)
  add_subdirectory(fan)
  list(APPEND EXTRA_LIBS fan)
  list(APPEND EXTRA_INCLUDES "${PROJECT_SOURCE_DIR}/fan")
endif()

# add the executable
add_executable(ActiveFanControl fan_controller.cpp)

set_target_properties(config PROPERTIES LINKER_LANGUAGE CXX)
set_target_properties(thermal PROPERTIES LINKER_LANGUAGE CXX)
set_target_properties(pwm PROPERTIES LINKER_LANGUAGE CXX)
set_target_properties(fan PROPERTIES LINKER_LANGUAGE CXX)

target_link_libraries(ActiveFanControl PUBLIC ${EXTRA_LIBS})

# add the binary tree to the search path for include files
target_include_directories(ActiveFanControl PUBLIC
                          "${PROJECT_BINARY_DIR}"
                          ${EXTRA_INCLUDES}
                          )
